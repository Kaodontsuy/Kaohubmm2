-- > Declarations < --

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LP = Players.LocalPlayer
local roles
local player = Players.LocalPlayer
local role = player:WaitForChild("Role")  -- Vai trò của người chơi, ví dụ: Murderer, Sheriff, Innocent
local buttonShoot = script.Parent:WaitForChild("ButtonShoot")  -- Nút bắn trong GUI
local buttonAim = script.Parent:WaitForChild("ButtonAim")  -- Nút Aim trong GUI

local aiming = false  -- Biến để theo dõi trạng thái ngắm
local Murder, Sheriff, Hero

-- > Functions <--

function CreateHighlight() -- make any new highlights for new players
	for i, v in pairs(Players:GetChildren()) do
		if v ~= LP and v.Character and not v.Character:FindFirstChild("Highlight") then
			Instance.new("Highlight", v.Character)           
		end
	end
end

function UpdateHighlights() -- Get Current Role Colors (messy)
	for _, v in pairs(Players:GetChildren()) do
		if v ~= LP and v.Character and v.Character:FindFirstChild("Highlight") then
			local Highlight = v.Character:FindFirstChild("Highlight")
			if v.Name == Sheriff and IsAlive(v) then
				Highlight.FillColor = Color3.fromRGB(0, 0, 225)
			elseif v.Name == Murder and IsAlive(v) then
				Highlight.FillColor = Color3.fromRGB(225, 0, 0)
			elseif v.Name == Hero and IsAlive(v) and not IsAlive(game.Players[Sheriff]) then
				Highlight.FillColor = Color3.fromRGB(255, 250, 0)
			else
				Highlight.FillColor = Color3.fromRGB(0, 225, 0)
			end
		end
	end
end	

function IsAlive(Player) -- Simple sexy function
	for i, v in pairs(roles) do
		if Player.Name == i then
			if not v.Killed and not v.Dead then
				return true
			else
				return false
			end
		end
	end
end

-- Tạo ESP cho người chơi
local function createESP()
    -- Tạo BillboardGui để hiển thị tên vai trò
    local esp = Instance.new("BillboardGui")
    esp.Adornee = player.Character
    esp.Size = UDim2.new(0, 100, 0, 50)
    esp.StudsOffset = Vector3.new(0, 3, 0)
    esp.Parent = player.PlayerGui
    
    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.TextColor3 = Color3.fromRGB(255, 255, 255)
    label.TextScaled = true
    label.Text = role.Value
    label.Parent = esp
end

-- Tạo chức năng bắn cho Murderer
local function shoot()
    if player.Character and player.Character:FindFirstChild("Humanoid") then
        local knife = player.Backpack:FindFirstChild("Knife")  -- Giả sử có một vũ khí tên là "Knife" cho Murderer
        if knife then
            -- Thực hiện hành động bắn (hoặc đâm) đối với Knife
            print("Murderer đã tấn công!")
            -- Thực hiện hành động giết người ở đây (ví dụ: tiêu diệt đối thủ)
        end
    end
end

-- Tạo chức năng ngắm cho Murderer (Aimbot)
local function aim()
    if role.Value == "Murderer" then
        aiming = true
        print("Murderer đang ngắm vào mục tiêu...")
        -- Thực hiện ngắm vào mục tiêu gần nhất
        local target = nil
        local closestDistance = math.huge

        -- Tìm kiếm mục tiêu gần nhất (không phải mình)
        for _, otherPlayer in pairs(game.Players:GetPlayers()) do
            if otherPlayer ~= player and otherPlayer.Character then
                local distance = (player.Character.HumanoidRootPart.Position - otherPlayer.Character.HumanoidRootPart.Position).Magnitude
                if distance < closestDistance then
                    closestDistance = distance
                    target = otherPlayer.Character
                end
            end
        end

        -- Nếu tìm thấy mục tiêu, nhắm vào họ
        if target then
            print("Đã tìm thấy mục tiêu: " .. target.Name)
            -- Có thể làm gì đó như thay đổi hướng camera hoặc hiển thị crosshair
        end
    else
        print("Chỉ Murderer mới có thể ngắm!")
    end
end

-- Tạo hành động cho Murderer khi nhấn Aim
buttonAim.MouseButton1Click:Connect(function()
    aim()  -- Khi nhấn nút Aim, sẽ gọi hàm aim()
end)

-- Tạo hành động bắn khi nhấn nút Shoot
buttonShoot.MouseButton1Click:Connect(function()
    if role.Value == "Sheriff" then
        shoot()  -- Chỉ cho phép Sheriff bắn
    elseif role.Value == "Murderer" and aiming then
        -- Nếu người chơi là Murderer và đang trong trạng thái ngắm, thực hiện hành động giết người
        print("Giết người! Murderer đã tấn công.")
        -- Thực hiện hành động giết người tại đây, ví dụ: hạ gục một người chơi hoặc tương tác với một mục tiêu.
    else
        print("Chỉ Murderer mới có thể giết người khi đang ngắm!")
    end
end)

-- Gọi hàm khi vai trò thay đổi
role.Changed:Connect(createESP)

-- Kiểm tra và xử lý vai trò
local function handleRole()
    if role.Value == "Murderer" then
        -- Hành động cho Murderer
        print("Bạn là Murderer!")
    elseif role.Value == "Sheriff" then
        -- Hành động cho Sheriff
        print("Bạn là Sheriff!")
    else
        -- Hành động cho Innocent
        print("Bạn là Innocent!")
    end
end

-- Gọi hàm xử lý vai trò khi script bắt đầu
handleRole()

-- Tạo chức năng di chuyển cho ButtonShoot
local function moveShootButton()
    local randomPosition = UDim2.new(math.random(), 0, math.random(), 0)  -- Di chuyển ngẫu nhiên trong GUI
    buttonShoot.Position = randomPosition
end

-- Di chuyển ButtonShoot sau mỗi 3 giây
while true do
    wait(3)
    moveShootButton()
end

-- > Loops < --

RunService.RenderStepped:connect(function()
	roles = ReplicatedStorage:FindFirstChild("GetPlayerData", true):InvokeServer()

			Murder = i
		elseif v.Role == 'Sheriff'then
			Sheriff = i
		elseif v.Role == 'Hero'then
			Hero = i
		end
	end
	CreateHighlight()
	UpdateHighlights()
end)
